#include <stdio.h>
#define size 5
int array[size];
int top = -1;
//push
void push(int value){
    if(top == size-1){
        printf("No space left\n");
    }
    else
    {
        top++;
        array[top] = value;
        printf("The value %d pushed to array \n",value);
    }
}
//peek  operation
void pop(){
    if(top == -1){
        printf("no value at stack");
    }
    else
    {
        printf("The top value is : %d\n", array[top]);
       
        top--;
    }
}

    if(top == -1){

}
