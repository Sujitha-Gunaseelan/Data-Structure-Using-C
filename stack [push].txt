// Online C compiler to run C program online
#include <stdio.h>
#define size 5
int array[size];
int top = -1;
//push
void push(int value){
    if(top == size-1){
        printf("No space left\n");
    }
    else
    {
        top++;
        array[top] = value;
        printf("The value %d pushed to array \n",value);
    }
}

int main() {
push(10);

    return 0;
}
